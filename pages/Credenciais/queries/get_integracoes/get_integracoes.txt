SELECT
  c.nome_empresa,
  c.cnpj,
  ca.software,
  ca.nome_aplicacao,
  cc.conn_label,             -- opcional: nome da conex√£o
  cc.id                      -- opcional: id da credencial
FROM public.credenciais_clientes cc
JOIN public.clients c
  ON c.id = cc.client_id
JOIN public.credenciais_aplicativos ca
  ON ca.id = cc.app_id
WHERE (
  {{ (tabClientes.selectedRow && tabClientes.selectedRow.id)
      ? `cc.client_id = ${tabClientes.selectedRow.id}`
      : '1=1' }}
)
ORDER BY c.nome_empresa, ca.software, cc.conn_label;
